<?php
error_reporting(E_ALL);
ini_set('display_errors', 1);

$file = __DIR__ . "/valeur.txt";
$valeur = "Aucune valeur reçue pour l'instant.";

// ======================
// TRAITEMENT POST
// ======================
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    if (isset($_POST['valeur'])) {
        $valeur = $_POST['valeur'];
        $result = file_put_contents($file, $valeur);

        if ($result === false) {
            echo "Erreur écriture fichier";
        } else {
            echo "Valeur enregistrée : " . htmlspecialchars($valeur);
        }
    } else {
        echo "POST reçu mais pas de valeur";
    }
    exit;
}

// ======================
// TRAITEMENT GET
// ======================
if (file_exists($file)) {
    $valeur = file_get_contents($file);
}
?>

<!DOCTYPE html>
<html lang="fr">
<head>
    <meta charset="UTF-8">
    <title>Valeur Arduino</title>

    <!-- Rafraîchissement toutes les 5 secondes -->
    <meta http-equiv="refresh" content="5">

    <style>
        body {
            font-family: Arial, sans-serif;
            text-align: center;
            margin-top: 50px;
        }
        .valeur {
            font-size: 2em;
            color: #007BFF;
            margin: 20px;
        }
        button {
            padding: 10px 20px;
            margin: 10px;
            font-size: 16px;
            cursor: pointer;
        }
    </style>
</head>
<body>

    <h1>Dernière valeur reçue de l'Arduino</h1>

    <div class="valeur">
        <?php echo htmlspecialchars($valeur); ?>
    </div>

    <button onclick="fetch('http://192.168.100.41/led')">
        Allumer LED
    </button>

    <button onclick="fetch('http://192.168.100.41/son')">
        Activer Buzzer
    </button>

</body>
</html>
